Примерно так может быть оформлена ваша функция в файле task3.py

    from io import StringIO
    import csv


    def task(csvString):
        f = StringIO(csvString)
        reader = csv.reader(f, delimiter=',')
        out = []
        for row in reader:
            out.append(row)
        return out
